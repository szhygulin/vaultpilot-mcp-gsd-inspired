```typescript
export const HANDLE_TTL_MS = 15 * 60 * 1000;       // 15 min
export type HandleStatus = "prepared" | "previewed" | "sent" | "cancelled";

export interface PrepareArgs {
  to: string;              // raw 0x-string as agent passed it — NOT normalized to checksum
  valueWei: string;        // raw decimal/hex string as agent passed it
}
export interface PreparedTx {
  chainId: number;
  to: Address;
  valueWei: bigint;
  data: Hex;
  nonce?: number;          // pinned at preview time
  gas?: bigint;            //   ditto
  maxFeePerGas?: bigint;   //   ditto
  maxPriorityFeePerGas?: bigint;   // ditto
}
export interface PreviewPinned {
  nonce: number;
  gas: bigint;
  maxFeePerGas: bigint;
  maxPriorityFeePerGas: bigint;
  previewToken: string;        // crypto.randomUUID()
  presignHash: Hex;            // keccak256(serializeTransaction(...))
  selector: Hex | null;        // first 4 bytes of data, or null if data === "0x"
}
export interface HandleRecord {
  handle: string;
  args: PrepareArgs;
  tx: PreparedTx;
  payloadFingerprint: Hex;
  status: HandleStatus;
  createdAt: number;
  pinned?: PreviewPinned;       // present iff status >= "previewed"
  sentAt?: number;
  txHash?: Hex;
  cancelledAt?: number;
}
```

## Error codes (research § Q15 — locked set)

```typescript
export type ErrorCode =
  | "WALLET_NOT_PAIRED"        // getStatus() returned null at prepare/preview/send
  | "HANDLE_NOT_FOUND"         // lookup miss
  | "HANDLE_EXPIRED"           // > 15min from createdAt
  | "WRONG_STATUS"             // illegal state transition
  | "PREVIEW_REQUIRED"         // send_transaction on prepared (not previewed) handle
  | "PREVIEW_TOKEN_MISMATCH"   // args.previewToken !== record.pinned.previewToken

  | "LEDGER_REJECTED"          // signClient.request rejected (WC code 5000 / "User rejected")
  | "BROADCAST_FAILED"         // signClient.request rejected (other reasons — nonce, underpriced, relay)

export interface StructuredError {

  message: string;

  cause?: string;
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Agent → MCP (prepare args) | Untrusted: `to` + `valueWei` could be agent-fabricated. `PREPARE RECEIPT` block surfaces them verbatim for user cross-check. |
| In-process state (handle store) | Trusted boundary within the MCP — but defended against accidental corruption (state-machine refuses illegal transitions; fingerprint re-check at send time catches mutation). |
