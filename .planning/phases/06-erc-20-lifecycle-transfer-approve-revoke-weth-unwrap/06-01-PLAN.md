## `parseAmountStrict` exact signature (LOCKED in this plan)

```typescript
// src/signing/amount.ts (NEW — DF-2 LOCKED placement)

import { parseUnits } from "viem";

// Distinct error class so callers can `instanceof`-match without coupling to
// a generic Error. Mirror the pattern at `src/signing/error-codes.ts:61`
// (StructuredError interface) — but for THROW-shape (parseAmountStrict throws;
// the caller catches and converts to `errEnvelope("INVALID_INPUT", ...)` at
// the tool boundary).
export class InvalidAmountError extends Error {
  readonly kind: "empty" | "format" | "fractional-overflow";
  constructor(message: string, kind: "empty" | "format" | "fractional-overflow") {
    super(message);
    this.name = "InvalidAmountError";
    this.kind = kind;
  }
}

/**
 * Strict pre-validation guard before `viem.parseUnits`.
 *
 * Rejects (each with a distinct error message naming the offending input):
 *   - empty string / whitespace-only      → InvalidAmountError(kind: "empty")
 *   - any character outside `[0-9.]`      → kind: "format" (catches "-1", "1e6", "1,000", "abc")
 *   - leading dot ".5" / trailing dot "5."→ kind: "format"
 *   - more than one dot                   → kind: "format"
 *   - fractional digits > decimals        → kind: "fractional-overflow"
 *     (e.g. "1.23456789" with decimals=6 — the load-bearing off-by-decimal
 *     guard the project CLAUDE.md "Decimal-aware arithmetic" rule names)
 *
 * Accepts and delegates to `parseUnits` for:
 *   - integer-only digit strings         "100"
 *   - decimal strings with ≤ decimals    "100.5", "1.123456" (decimals=6)
 *   - "0"
 *
 * The delegation invariant: for every accepted input X with decimals D,
 * `parseAmountStrict(X, D) === parseUnits(X, D)` — the strict guard is a
 * pre-filter, never a re-implementation.
 *
 * @throws InvalidAmountError on any rejected input.
 */
export function parseAmountStrict(amountStr: string, decimals: number): bigint;
```
