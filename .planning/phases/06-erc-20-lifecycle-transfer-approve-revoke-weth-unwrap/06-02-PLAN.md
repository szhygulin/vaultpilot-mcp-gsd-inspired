// All exports VERIFIED via empirical Bash probe (research § Topic 1, § Code Example 1).
import {
  encodeFunctionData,
  decodeFunctionData,
  erc20Abi,
  parseAbi,
  type Address,
  type Hex,
} from "viem";

// Selector table — exported so preview_send.ts can switch on it without
// duplicating slice(0,10) magic. Format-fanout-sentinel rule: never inline.
// Verified selectors via keccak(toHex("transfer(address,uint256)")).slice(0,10):
//   transfer → 0xa9059cbb
//   approve  → 0x095ea7b3
export const ERC20_SELECTORS = {
  transfer: "0xa9059cbb" as Hex,
  approve:  "0x095ea7b3" as Hex,
} as const;

// PREP-29 unlimited-approval sentinel. Equal to viem.maxUint256 (verified
// research § Topic 4: viem.maxUint256 === (2n ** 256n - 1n)).
export const MAX_UINT256: bigint = (1n << 256n) - 1n;

// WETH9 fragment for combined-ABI decode in preview_send. (Plan 06-04's
// prepare_weth_unwrap re-imports the ABI from src/protocols/weth9.ts; the
// fragment here is for DECODE only so preview_send.ts has a single
// combined-ABI decode path. Plan 06-04 may consolidate.)
const WETH9_DECODE_FRAGMENT = parseAbi(["function withdraw(uint256 amount)"]);

// Combined ABI for preview-time decode. preview_send dispatches over
// `decoded.functionName`; never trusts the agent's claim about what the
// data is.
export const ERC20_COMBINED_DECODE_ABI = [
  ...erc20Abi,
  ...WETH9_DECODE_FRAGMENT,
] as const;

export function encodeErc20Transfer(to: Address, amount: bigint): Hex {
  return encodeFunctionData({
    abi: erc20Abi,
    functionName: "transfer",
    args: [to, amount],
  });
}

export function encodeErc20Approve(spender: Address, amount: bigint): Hex {
  return encodeFunctionData({
    abi: erc20Abi,
    functionName: "approve",
    args: [spender, amount],
  });
}

// Discriminated union — preview_send switches on `.kind`. Mirror the
// FourbyteResult shape from src/clients/fourbyte.ts (PATTERNS.md line 165).
export type Erc20Decoded =
  | { kind: "transfer"; to: Address; amount: bigint }
  | { kind: "approve"; spender: Address; amount: bigint; isUnlimited: boolean }
  | { kind: "withdraw"; amount: bigint }
  | { kind: "unknown"; selector: Hex };

export function decodeErc20Call(data: Hex): Erc20Decoded {
  // selector === first 4 bytes; if data === "0x" return unknown
  if (data === "0x" || data.length < 10) {
    return { kind: "unknown", selector: data as Hex };
  }
  const selector = data.slice(0, 10) as Hex;
  try {
    const decoded = decodeFunctionData({
      abi: ERC20_COMBINED_DECODE_ABI,
      data,
    });
    switch (decoded.functionName) {
      case "transfer": {
        const [to, amount] = decoded.args as readonly [Address, bigint];
        return { kind: "transfer", to, amount };
      }
      case "approve": {
        const [spender, amount] = decoded.args as readonly [Address, bigint];
        return { kind: "approve", spender, amount, isUnlimited: amount === MAX_UINT256 };
      }
      case "withdraw": {
        const [amount] = decoded.args as readonly [bigint];
        return { kind: "withdraw", amount };
      }
      default:
        return { kind: "unknown", selector };
    }
  } catch {
    // Unknown selector → AbiFunctionSignatureNotFoundError. Surface as
    // unknown so the 4byte block (existing — Plan 04-05) still fires.
    return { kind: "unknown", selector };
  }
}

// ESM spy-affordance per CLAUDE.md "ESM spy-affordance indirection"
// convention (codified in PR #26). preview_send.ts imports `_protocols`
// and calls `_protocols.decodeErc20Call(data)` so tests can vi.spyOn the
// decode without monkey-patching the production module import path.
export const _protocols = { decodeErc20Call };
```

// Wide eth_call preview-time simulation helper. Research § Topic 9 (DF-1
// LOCKED Option B). Lives on the src/signing/ shelf alongside other
// preview-time concerns. Consumed by preview_send.ts; covers ALL tx shapes
// including native sends (defense-in-depth uniform per the Phase 4 PREP-07
// schema-gate lifted-to-all-tools precedent).

import type { Address, Hex } from "viem";
import { call } from "viem/actions";
import type { PublicClient } from "viem";

export type SimulationStatus = "ok" | "revert" | "error";

export interface SimulationResult {
  status: SimulationStatus;
  resultData: Hex | null;   // call return data on ok; null on revert/error
  errorMessage: string | null;
}

export async function runPreviewSimulation(input: {
  client: PublicClient;
  sender: Address;
  tx: { to: Address; valueWei: bigint; data: Hex };
}): Promise<SimulationResult> {
  try {
    const result = await call(input.client, {
      account: input.sender,
      to: input.tx.to,
      value: input.tx.valueWei,
      data: input.tx.data,
    });
    return { status: "ok", resultData: (result.data ?? "0x") as Hex, errorMessage: null };
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    // viem signals reverts via the error message containing "revert" /
    // "execution reverted". Operational errors (RPC down, timeout) hit
    // the same catch; distinguish via message text. Both are non-blocking
    // for the trust anchor (the user still sees the hash and decides).
    if (/revert|execution reverted/i.test(message)) {
      return { status: "revert", resultData: null, errorMessage: message };
    }
    return { status: "error", resultData: null, errorMessage: message };
  }
}
```
