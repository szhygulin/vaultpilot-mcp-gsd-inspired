import { ChainId, ChainName, chainIdFromName, chainNameFromId } from "../config/contracts.js";
import { getChainClient, isPublicNodeFallback } from "../chains/registry.js";
import { scanErc20Balances } from "../chains/erc20-scanner.js";
import { loadTokenRegistry } from "../tokens/registry.js";
import { getPrices } from "../pricing/defillama.js";

const ALL_CHAINS: readonly ChainName[] = ["ethereum", "arbitrum", "polygon", "base", "optimism"];
const PER_CHAIN_TIMEOUT_MS = 10_000;  // research § A9 — viem default may exceed MCP response budget

interface ChainPortfolio {
  chain: ChainName;
  nativeBalance: NativeBalanceRow;     // existing Phase 2 shape; gains explicit `chain` field
  erc20Balances: Erc20BalanceRow[];    // existing Phase 2 shape; per-row `chain` field added
  totalUsd: string;
  rpcDegraded?: boolean;
}

interface CrossChainPortfolioResult {
  perChain: Partial<Record<ChainName, ChainPortfolio>>;
  chainErrors: Array<{ chain: ChainName; reason: string }>;
  totalUsd: string;  // sum of perChain[*].totalUsd
}

async function readChainPortfolioWithTimeout(
  chainName: ChainName,
  wallet: Address,
  dustThreshold: bigint,
): Promise<ChainPortfolio> {
  const abort = new AbortController();
  const timeout = setTimeout(() => abort.abort(), PER_CHAIN_TIMEOUT_MS);
  try {
    return await Promise.race([
      readChainPortfolio(chainName, wallet, dustThreshold),
      new Promise<ChainPortfolio>((_, reject) =>
        abort.signal.addEventListener("abort", () =>
          reject(new Error(`timeout after ${PER_CHAIN_TIMEOUT_MS}ms`))),
      ),
    ]);
  } finally {
    clearTimeout(timeout);
  }
}

async function readChainPortfolio(
  chainName: ChainName,
  wallet: Address,
  dustThreshold: bigint,
): Promise<ChainPortfolio> {
  const chainId = chainIdFromName(chainName);
  const client = getChainClient(chainId);
  const [nativeBalanceRaw, erc20Balances] = await Promise.all([
    client.getBalance({ address: wallet }),
    scanErc20Balances(wallet, chainId),
  ]);
  // ... existing per-chain logic from Plan 08-02 — per-chain pricing via getPrices([{chain, address}]) ...
  const priceCoins: PriceCoin[] = erc20Balances.map((r) => ({ chain: chainName, address: r.address }));
  // ... aggregate USD ...
  return {
    chain: chainName,
    nativeBalance: { /* with chain field */ },
    erc20Balances: erc20Balances.map((r) => ({ ...r, chain: chainName })),
    totalUsd,
    ...(isPublicNodeFallback(chainId) ? { rpcDegraded: true } : {}),
  };
}

// Handler body branch (LOCKED):
async function handler(args: GetPortfolioSummaryArgs): Promise<ToolResponse> {
  // ... existing input validation ...

  if (args.chain !== undefined) {
    // SINGLE-CHAIN branch — existing Plan 08-02 logic, BYTE-FROZEN at the per-chain layer.
    const result = await readChainPortfolioWithTimeout(args.chain as ChainName, wallet, dustThreshold);
    return {
      content: [{ type: "text", text: renderSingleChainText(result) }],
      structuredContent: result,
    };
  }

  // CROSS-CHAIN branch — Plan 08-03 fan-out.
  const results = await Promise.allSettled(
    ALL_CHAINS.map((c) => readChainPortfolioWithTimeout(c, wallet, dustThreshold)),
  );

  const perChain: Partial<Record<ChainName, ChainPortfolio>> = {};
  const chainErrors: Array<{ chain: ChainName; reason: string }> = [];
  let totalUsdNum = 0;

  for (let i = 0; i < results.length; i++) {
    const chain = ALL_CHAINS[i]!;
    const result = results[i]!;
    if (result.status === "fulfilled") {
      perChain[chain] = result.value;
      totalUsdNum += parseFloat(result.value.totalUsd);
    } else {
      chainErrors.push({
        chain,
        reason: result.reason instanceof Error ? result.reason.message : String(result.reason),
      });
    }
  }

  const totalUsd = totalUsdNum.toFixed(2);
  return {
    content: [{ type: "text", text: renderCrossChainText({ perChain, chainErrors, totalUsd }) }],
    structuredContent: { perChain, chainErrors, totalUsd },
  };
}
