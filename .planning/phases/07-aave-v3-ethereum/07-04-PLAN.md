// src/clients/etherscan.ts (NEW)
//
// Etherscan V2 unified-API client. Surface used by check_contract_security
// (Plan 07-04). Best-effort never-throws contract; mirror of
// src/clients/fourbyte.ts shape.
//
// Etherscan V2 endpoint:
//   https://api.etherscan.io/v2/api?chainid={N}&apikey={KEY}&module={M}&action={A}&...
//
// Free tier rate limits (verified 2026-05-13 via web search):
//   - 5 calls/sec
//   - 100k calls/day
//   - ONE API key works across ALL supported chains (Phase 8 inherits)
//
// Per-session rate limit (research § Topic 7): 5 calls per agent session,
// soft refusal with `{ kind: "rate-limited", ... }` envelope when exceeded.
// Counter increments BEFORE the network call; cached hits do NOT consume.

import { type Address } from "viem";

import { log } from "../diagnostics/logger.js";

const ETHERSCAN_API_URL = "https://api.etherscan.io/v2/api";
const ETHERSCAN_TIMEOUT_MS = 3000;  // 2× fourbyte (Etherscan latency higher)
const CACHE_MAX_ENTRIES = 256;
const PER_SESSION_CALL_LIMIT = 5;

export type EtherscanResult =
  | { kind: "not-applicable" }
  | { kind: "ok"; verified: boolean; proxy: boolean; implementation?: Address; contractName: string; compilerVersion: string; creatorAddress: Address; creationTxHash: string; creationTimestamp: number; ageDays: number | "unknown"; privilegedFunctions: string[]; accessControlMarkers: string[]; abi?: string }
  | { kind: "not-verified" }
  | { kind: "rate-limited"; message: string }
  | { kind: "error"; message: string };

const cache = new Map<Address, EtherscanResult>();
let agentSessionCallCount = 0;

/**
 * Best-effort contract security probe. Never throws; returns one of the 5
 * discriminated-union arms. `null` address → `not-applicable` (no network
 * call; defensive guard for caller-passes-null).
 *
 * Two parallel Etherscan V2 calls via Promise.all:
 *   - getsourcecode → verified-source flag + ABI + Proxy/Implementation + compiler version + contract name
 *   - getcontractcreation → creator address + creation block + timestamp + creation tx hash
 *
 * If either call fails, the result is the `error` arm with the failing call's
 * message. If both succeed but the source is unverified, result is `not-
 * verified`.
 *
 * Privileged-role heuristic (research § Topic 7): parse ABI, scan for
 * functions matching the privileged-name set, scan for AccessControl
 * markers. Surface as TWO arrays — agent reads both.
 */
export async function checkContractSecurity(address: Address | null, apiKey: string): Promise<EtherscanResult> {
  if (address === null) return { kind: "not-applicable" };

  const cached = cache.get(address);
  if (cached) return cached;

  // Increment BEFORE the network call. Cached hits above never reach here.
  if (agentSessionCallCount >= PER_SESSION_CALL_LIMIT) {
    return {
      kind: "rate-limited",
      message: `per-session limit (${PER_SESSION_CALL_LIMIT} calls) exceeded; resets at MCP server restart. Free Etherscan tier allows 100k/day; raise via paid plan if needed.`,
    };
  }
  agentSessionCallCount += 1;

  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), ETHERSCAN_TIMEOUT_MS);

  let result: EtherscanResult;
  try {
    const sourceUrl = `${ETHERSCAN_API_URL}?chainid=1&apikey=${apiKey}&module=contract&action=getsourcecode&address=${address}`;
    const creationUrl = `${ETHERSCAN_API_URL}?chainid=1&apikey=${apiKey}&module=contract&action=getcontractcreation&contractaddresses=${address}`;
    const [sourceResp, creationResp] = await Promise.all([
      fetch(sourceUrl, { signal: controller.signal }),
      fetch(creationUrl, { signal: controller.signal }),
    ]);
    if (!sourceResp.ok || !creationResp.ok) {
      const status = sourceResp.ok ? creationResp.status : sourceResp.status;
      result = { kind: "error", message: `Etherscan V2 returned HTTP ${status}` };
    } else {
      const sourceBody = (await sourceResp.json()) as { status: string; result: Array<Record<string, string>> };
      const creationBody = (await creationResp.json()) as { status: string; result: Array<Record<string, string>> };
      // Etherscan returns status "1" on success, "0" on error. Result array
      // empty or contains "Contract source code not verified" for unverified.
      if (sourceBody.status !== "1" || !sourceBody.result || sourceBody.result.length === 0) {
        result = { kind: "error", message: `Etherscan getsourcecode failed: ${JSON.stringify(sourceBody.result)}` };
      } else {
        const src = sourceBody.result[0];
        const verified = src.SourceCode !== "" && src.SourceCode !== "Contract source code not verified";
        if (!verified) {
          result = { kind: "not-verified" };
        } else {
          // Parse the ABI for privileged-role enumeration.
          let privilegedFunctions: string[] = [];
          let accessControlMarkers: string[] = [];
          try {
            const abi = JSON.parse(src.ABI) as Array<{ type?: string; name?: string; inputs?: Array<{ type: string }> }>;
            // Privileged-name set per research § Topic 7.
            const PRIVILEGED_NAMES = ["upgradeTo", "upgradeToAndCall", "setAdmin", "transferOwnership", "renounceOwnership", "setImplementation", "pause", "unpause", "mint", "burn", "blacklist", "freeze"];
            const ACCESS_CONTROL_NAMES = ["hasRole", "getRoleAdmin", "grantRole", "revokeRole", "renounceRole", "DEFAULT_ADMIN_ROLE", "paused"];
            for (const fn of abi) {
              if (fn.type !== "function" || !fn.name) continue;
              const sig = `${fn.name}(${(fn.inputs ?? []).map((i) => i.type).join(",")})`;
              if (PRIVILEGED_NAMES.includes(fn.name)) privilegedFunctions.push(sig);
              if (ACCESS_CONTROL_NAMES.includes(fn.name)) accessControlMarkers.push(sig);
            }
          } catch {
            // ABI parse failure — surface empty arrays; verified flag stays true.
          }
          // Creation data.
          let creatorAddress = "0x0000000000000000000000000000000000000000" as Address;
          let creationTxHash = "0x";
          let creationTimestamp = 0;
          if (creationBody.status === "1" && creationBody.result && creationBody.result.length > 0) {
            const c = creationBody.result[0];
            creatorAddress = (c.contractCreator ?? "0x0") as Address;
            creationTxHash = c.txHash ?? "0x";
            creationTimestamp = Number(c.timestamp ?? 0);
          }
          // Age computation with defensive surfacing for very-old contracts.
          const ageDays: number | "unknown" = creationTimestamp === 0
            ? "unknown"
            : Math.floor((Date.now() / 1000 - creationTimestamp) / 86400);
          result = {
            kind: "ok",
            verified: true,
            proxy: src.Proxy === "1",
            implementation: src.Implementation !== "" && src.Implementation !== "0x" ? (src.Implementation as Address) : undefined,
            contractName: src.ContractName ?? "",
            compilerVersion: src.CompilerVersion ?? "",
            creatorAddress,
            creationTxHash,
            creationTimestamp,
            ageDays,
            privilegedFunctions,
            accessControlMarkers,
            abi: src.ABI,
          };
        }
      }
    }
  } catch (err) {
    const errorObj = err as Error;
    if (errorObj?.name === "AbortError") {
      result = { kind: "error", message: `Etherscan V2 unreachable (timeout ${ETHERSCAN_TIMEOUT_MS}ms)` };
    } else {
      result = { kind: "error", message: `Etherscan V2 unreachable: ${errorObj?.message ?? String(err)}` };
    }
    log("warn", `Etherscan V2 lookup failed for ${address}: ${result.message}`);
  } finally {
    clearTimeout(timer);
  }

  cacheInsert(address, result);
  return result;
}

function cacheInsert(address: Address, result: EtherscanResult): void {
  if (cache.size >= CACHE_MAX_ENTRIES) {
    const oldestKey = cache.keys().next().value;
    if (oldestKey !== undefined) cache.delete(oldestKey);
  }
  cache.set(address, result);
}

export function _resetEtherscanCacheForTesting(): void {
  cache.clear();
}

export function _resetEtherscanRateCounterForTesting(): void {
  agentSessionCallCount = 0;
}
```

    if (!/^0x[0-9a-fA-F]{40}$/.test(rawAddress)) {
      return { isError: true, content: [...], structuredContent: errEnvelope("INVALID_INPUT", "address must be a 0x-prefixed 40-hex-character Ethereum address") };
    }
    const address = getAddress(rawAddress);  // EIP-55 normalize

    const apiKey = getEtherscanApiKey();
    if (apiKey === undefined) {
      return {
        isError: true,
        content: [{ type: "text", text: "error: ETHERSCAN_API_KEY not set. Register at https://etherscan.io/apis (free; 5 req/sec, 100k req/day; one key works across all chains)." }],
        structuredContent: errEnvelope("INTERNAL_ERROR", "ETHERSCAN_API_KEY not set", "register at https://etherscan.io/apis"),
      };
    }

    const result = await checkContractSecurity(address, apiKey);
    if (result.kind === "rate-limited") {
      return { isError: true, content: [{ type: "text", text: `error: ${result.message}` }], structuredContent: errEnvelope("INTERNAL_ERROR", result.message, "rate-limit") };
    }
    if (result.kind === "error") {
      return { isError: true, content: [{ type: "text", text: `error: ${result.message}` }], structuredContent: errEnvelope("INTERNAL_ERROR", result.message, "etherscan-unreachable") };
    }
    if (result.kind === "not-verified") {
      return { content: [{ type: "text", text: `${address}: NOT verified on Etherscan.` }], structuredContent: { chain: "ethereum" as const, address, verified: false } };
    }
    if (result.kind === "not-applicable") {
      // Should not happen — defensive.
      return { content: [{ type: "text", text: "not-applicable" }], structuredContent: { chain: "ethereum" as const, address, verified: false } };
    }
    // kind === "ok"
    const summary = [
      `${address}: verified${result.proxy ? " (proxy)" : ""}; ${result.contractName}; compiler ${result.compilerVersion}; age ${result.ageDays} days`,
      result.proxy && result.implementation ? `  implementation: ${result.implementation}` : null,
      result.privilegedFunctions.length > 0 ? `  privileged: ${result.privilegedFunctions.join(", ")}` : null,
      result.accessControlMarkers.length > 0 ? `  accessControl: ${result.accessControlMarkers.join(", ")}` : null,
    ].filter((s) => s !== null).join("\n");

    return {
      content: [{ type: "text", text: summary }],
      structuredContent: {
        chain: "ethereum" as const,
        address,
        verified: true,
        proxy: result.proxy,
        implementation: result.implementation,
        contractName: result.contractName,
        compilerVersion: result.compilerVersion,
        creatorAddress: result.creatorAddress,
        creationTxHash: result.creationTxHash,
        ageDays: result.ageDays,
        privilegedFunctions: result.privilegedFunctions,
        accessControlMarkers: result.accessControlMarkers,
      },
    };
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    return { isError: true, content: [{ type: "text", text: `error: check_contract_security failed: ${message}` }], structuredContent: errEnvelope("INTERNAL_ERROR", "check_contract_security failed", message) };
  }
});
```
